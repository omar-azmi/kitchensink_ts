Interface HexStringReprConfig

configuration for customizing the hex-string representation made by hexStringOfArray.

the default configuration is:

const default_HexStringReprConfig: HexStringReprConfig = {
	sep: ", ",
	prefix: "0x",
	postfix: "",
	trailingSep: false,
	bra: "[",
	ket: "]",
	toUpperCase: true,
	radix: 16,
}
interface HexStringReprConfig {

    postfix: string;
    trailingSep: boolean;
    bra: string;
    ket: string;
    toUpperCase: boolean;
    radix: number;
}
